root/

engine/					// The runtime code that can be used by games, tools, and editor.

  system/			 	// OS primitives
    files/				// File read/write, directory traversal, file watching, timestamps.
    threads/			// Threads, mutexes, atomics, semaphores, thread-local storage.
    time/				// High-resolution timers, clocks, frame timing helpers.
	process/			// Launching external tools, process handles, exit codes.
	libarary/			// Dynamic library loading, DLL reload, symbol lookup.
	paths/				// Executable path, project path, content path, temp path helpers.
	console/			// Console output, key input for bootstrap tools, terminal helpers.

	This layer is usually static and resident.

  base/					// Basic types, macros, platform defines, assertions, common helpers  
	string/				// string output and formatting.
	character/			// character utility wrappers
	assert/

	memory/				// Allocators, arenas, memory tags, allocation apis
	containers/			// Arrays, hash, pools queues, string builders.
	strings/			// String views, formatting, UTF, parsing.
	math/				// Vectors, matrices, quaternions, geometry helpers.
	debug/				// Debug checks, crash handling, profiling markers, debug only utils.
	logging/			// Core loggin api used by all engine modules.
	config/				// Reading config values, command-line config, engine settings.

  
  app/					// Application shell services. Windows, OS events, and lifecycle.
    application/		// app startup, app state, command-line handoff, platform itegration		
	window/				// Window creation, resize, fullscreen, DPI, platform window handles.
    events/				// OS event pump, app events, window events, quit events
	input_source/		// Raw OS input collection before it becomes engine input.
    lifecycle/			// Startup, shutdown, suspend, resume, focus changes.

/* host responsibility */

	main()
	command-line parsing
	startup/shutdown sequence
	module loading
	service creation
	hot reload loop
	application/game/editor/tool loop


/* editor host */

	int main(int argc, char** argv)
	{
		engine_init();
	
		editor_host_desc_t desc = {};
		desc.argc = argc;
		desc.argv = argv;
	
		editor_host_run(&desc);
	
		engine_shutdown();
		return 0;
	}

/* game project running */

	game.exe
	loads project file
	reads module list
	starts engine runtime
	loads render/audio/physics/assets
	loads project game DLL
	enters game loop

/* game tool concept */

	Then each tool is a module or command registered with the tool runtime.
	
	tool.exe reflection-gen --project projects/sample_game
	tool.exe asset-compile --project projects/sample_game
	tool.exe shader-compile --project projects/sample_game
	tool.exe package --project projects/sample_game
	
	tool.exe reflection-gen
	tool.exe shader-compile
	tool.exe asset-compile
	tool.exe package
	
	int main(int argc, char** argv)
	{
		tool_runtime_init();
	
		register_tool_command("reflection-gen", reflection_gen_main);
		register_tool_command("shader-compile", shader_compiler_main);
		register_tool_command("asset-compile", asset_compiler_main);
		register_tool_command("package", packager_main);
	
		int result = tool_dispatch(argc, argv);
	
		tool_runtime_shutdown();
		return result;
	}
	
	tool.exe is the host
	reflection_gen is a tool module
	asset_compiler is a tool module
	shader_compiler is a tool module
	packager is a tool module

/* usage */

editor.exe --project my_game.project
  loads my_game.dll
  loads my_game_editor.dll
  loads engine runtime modules

game.exe --project my_game.project
  loads my_game.dll
  loads engine runtime modules

tool.exe reflection-gen --project my_game.project
  reads project source and generates code

tool.exe asset-compile --project my_game.project
  reads project content and writes cooked assets
