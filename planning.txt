organization layout

engine/					// The runtime code that can be used by games, tools, and editor.

  system/			 	// OS primitives
    files/				// File read/write, directory traversal, file watching, timestamps.
    threads/			// Threads, mutexes, atomics, semaphores, thread-local storage.
    time/				// High-resolution timers, clocks, frame timing helpers.
	process/			// Launching external tools, process handles, exit codes.
	libarary/			// Dynamic library loading, DLL reload, symbol lookup.
	paths/				// Executable path, project path, content path, temp path helpers.
	console/			// Console output, key input for bootstrap tools, terminal helpers.


	This layer is usually static and resident.

  base/ 				// Lowest level, no OS, no graphics, no assumptions
	base/				// Basic types, macros, platform defines, assertions, common helpers
	memory/				// Allocators, arenas, memory tags, allocation apis
	containers/			// Arrays, hash, pools queues, string builders.
	strings/			// String views, formatting, UTF, parsing.
	math/				// Vectors, matrices, quaternions, geometry helpers.
	debug/				// Debug checks, crash handling, profiling markers, debug only utils.
	logging/			// Core loggin api used by all engine modules.
	config/				// Reading config values, command-line config, engine settings.

  
  app/					// Application shell services. Windows, OS events, and lifecycle.
    application/		// app startup, app state, command-line handoff, platform itegration		
	window/				// Window creation, resize, fullscreen, DPI, platform window handles.
    events/				// OS event pump, app events, window events, quit events
	input_source/		// Raw OS input collection before it becomes engine input.
    lifecycle/			// Startup, shutdown, suspend, resume, focus changes.
